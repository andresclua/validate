isCheckbox() — Basic

Validates that at least one checkbox is checked.

import { isCheckbox } from '@andresclua/validate'

const checkboxes = document.querySelectorAll('input[name="interests"]')
const result = isCheckbox({
    elements: checkboxes,
    config: { minRequired: 1 }
})

isNumber() — Custom Validation Function

Add custom logic on top of built-in numeric validation.

import { isNumber } from '@andresclua/validate'

const result = isNumber({
    element: input.value,
    config: {
        required: true,
        customValidation: (num) => ({
            isValid: num % 2 === 0,
            errorMessage: 'The number must be even'
        })
    }
})

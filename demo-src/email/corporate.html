isEmail() — Corporate Domains

Rejects personal email providers (Gmail, Yahoo, Outlook). Only accepts company domain emails.

import { isEmail } from '@andresclua/validate'

const result = isEmail({
    element: input.value,
    config: { required: true, type: 'corporate' }
})

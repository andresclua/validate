Advanced — Form Validation

Validate multiple fields together and handle a form submission event.

import { isEmail, isString } from '@andresclua/validate'

const form = document.getElementById('my-form')

form.addEventListener('submit', (e) => {
    e.preventDefault()

    const nameResult = isString({
        element: form.querySelector('#name').value,
        config: { required: true, minLength: 2 }
    })

    const emailResult = isEmail({
        element: form.querySelector('#email').value,
        config: { required: true }
    })

    if (nameResult.isValid && emailResult.isValid) {
        console.log('Form valid — submit to server')
    } else {
        if (!nameResult.isValid) console.log('Name error:', nameResult.errorMessage)
        if (!emailResult.isValid) console.log('Email error:', emailResult.errorMessage)
    }
})

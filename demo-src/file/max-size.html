isFile() — Max Size

Reject files that exceed a maximum file size.

import { isFile } from '@andresclua/validate'

const result = isFile({
    element: input.files[0],
    config: {
        required: true,
        maxSize: 1 * 1024 * 1024,
        customMessage: {
            size: 'File must be smaller than 1MB'
        }
    }
})
